        <head>

        <!-- Кнопка вверх -->
        <style>
        .topNubex {
	        position: fixed;
	        right: 45px;
	        bottom: 45px;
        }
        
        </style>
        </head>

        <body>

        <!-- Кнопка вверх -->
        <a href="#" title="Вернуться к началу страницы" class="topNubex">Вверх\</a>

        </body>
